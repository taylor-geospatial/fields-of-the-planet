# Code Example

```python
import rasterio
import numpy as np

def load_tile(path: str, bands: list[int]) -> np.ndarray:
    """Load specific bands from a raster tile."""
    with rasterio.open(path) as src:
        data = src.read(bands)
    return data.astype(np.float32)

tiles = [load_tile(f"scene_{i}.tif", [4, 3, 2]) for i in range(10)]
mosaic = np.concatenate(tiles, axis=0)
```
